### 对方法加锁，保证只能同一个线程进行访问

- 使用信号量，效率较高
  ```swift
  let lock = DispatchSemaphore(value: 1)
  // 进入锁
  lock.wait()
  // 释放锁
  lock.signal()
  ```
- objc_sync_enter、objc_sync_exit（其实是 OC 里面的`@synchronized`），性能最差
  ```swift
  objc_sync_enter(object)
  objc_sync_exit(object)
  ```
- 使用 NSLock（实际是封装了一下 pthread_mutex）。性能中等
  ```swift
  let lock = NSLock()
  lock.lock()
  lock.unlock()
  ```
- 串行队列的同步任务

### 多读单写

- 使用栅栏函数。
- pthread_rwlock

**栅栏函数**
使用一个并行队列，读使用异步任务，写使用栅栏函数。这里使用并行队列的原因是因为如果串行队列，多读的时候就是串行的，影响效率

```swift
let concurrentQueue = DispatchQueue(label: "concurrentQueue", attributes: .concurrent)

/**
  使用栅栏函数使用多读单写
 */
func readFile() -> String {
  // 这里使用同步任务，阻塞进入的线程，保证即读即得
  concurrentQueue.sync {
    // 读写文件
  }
}

func writeFile() {
    // 这里使用异步任务，因为存入后不需要及时得到反馈结果
    concurrentQueue.async(flags: [.barrier]) {
    }
}
```

**pthread_rwlock**

```swift
class FileTest: XCTestCase {

  // 文件读写锁
  var pthreadRwlock = pthread_rwlock_t()


  func initRwlock() {
    pthread_rwlock_init(&pthreadRwlock, nil)
  }

  func readFile() -> String {
    pthread_rwlock_rdlock(&pthreadRwlock)
    defer {
      pthread_rwlock_unlock(&pthreadRwlock)
    }
    // 读文件
    return ""
  }

  func writeFile() {
    pthread_rwlock_wrlock(&pthreadRwlock)
    defer {
      pthread_rwlock_unlock(&pthreadRwlock)
    }
    // 写文件
  }

  deinit {
    pthread_rwlock_destroy(&pthreadRwlock)
  }
}

```
